# 🚀 Project Checkpoint: Health Trend Tracker SaaS

We are building a **Health Report Analyzer** (SaaS Model: ₹150/year).

```python
import os
import sys
from sqlalchemy import create_engine
from sqlalchemy.orm import sessionmaker
from dotenv import load_dotenv

load_dotenv()

DATABASE_URL = os.environ.get("DATABASE_URL", "")

# Fix Driver Prefix
if DATABASE_URL.startswith("postgresql://"):
    DATABASE_URL = DATABASE_URL.replace("postgresql://", "cockroachdb://", 1)

# Windows SSL Fix (Locates root.crt in AppData)
if sys.platform == "win32" and "sslrootcert" not in DATABASE_URL:
    appdata = os.environ.get("APPDATA")
    if appdata:
        cert_path = os.path.join(appdata, "postgresql", "root.crt")
        DATABASE_URL += f"&sslrootcert={cert_path}"

engine = create_engine(DATABASE_URL, pool_pre_ping=True)
SessionLocal = sessionmaker(autocommit=False, autoflush=False, bind=engine)

def get_db():
    db = SessionLocal()
    try:
        yield db
    finally:
        db.close()
```

#### 5\. `extractor.py` (Gemini 2.5 Logic + Prompt Engineering)

```python
from google import genai
from google.genai import types
import os
from schemas import ExtractionResult

client = genai.Client(api_key=os.environ.get("GEMINI_API_KEY"))

def call_gemini_model(model_name: str, image_bytes: bytes, mime_type: str):
    prompt_text = """
    Extract blood test data.
    CRITICAL RULES FOR REFERENCE RANGES:
    1. Standard Range ("10-20"): Set min_ref=10, max_ref=20.
    2. Upper Limit Only ("< 200"): Set min_ref=0, max_ref=200.
    3. Lower Limit Only ("> 55"): Set min_ref=55, max_ref=null.
    Rate confidence (0-100).
    """
    response = client.models.generate_content(
        model=model_name,
        contents=[
            types.Content(
                role="user",
                parts=[
                    types.Part(text=prompt_text),
                    types.Part(inline_data=types.Blob(data=image_bytes, mime_type=mime_type))
                ]
            )
        ],
        config={"response_mime_type": "application/json", "response_schema": ExtractionResult}
    )
    return response.parsed

def smart_extract(image_bytes: bytes, mime_type: str):
    valid_mime = mime_type if mime_type in ["application/pdf", "image/png"] else "image/jpeg"
    print(f"⚡ Attempting Gemini 2.5 Flash...")
    try:
        data = call_gemini_model("gemini-2.5-flash", image_bytes, valid_mime)
        model_used = "gemini-2.5-flash"
        
        if data.results:
            avg = sum(r.confidence_score for r in data.results) / len(data.results)
        else:
            avg = 0

        if avg < 75 or len(data.results) == 0:
            print(f"⚠️ Ambiguity. Escalating to Gemini 2.5 Pro...")
            data = call_gemini_model("gemini-2.5-pro", image_bytes, valid_mime)
            model_used = "gemini-2.5-pro"

    except Exception as e:
        print(f"❌ Flash Failed: {e}. Retrying with Pro...")
        data = call_gemini_model("gemini-2.5-pro", image_bytes, valid_mime)
        model_used = "gemini-2.5-pro"

    return data, model_used
```

#### 6\. `main.py` (FastAPI Server)

```python
from fastapi import FastAPI, UploadFile, File, Depends, HTTPException
from sqlalchemy.orm import Session
from database import engine, get_db
from models import Base, PatientReport, TestResult
from extractor import smart_extract
import uvicorn

Base.metadata.create_all(bind=engine)
app = FastAPI(title="Health AI")

@app.post("/analyze")
async def analyze(file: UploadFile = File(...), db: Session = Depends(get_db)):
    if file.content_type not in ["application/pdf", "image/jpeg", "image/png"]:
        raise HTTPException(400, detail="Invalid file type")
    
    file_bytes = await file.read()
    
    try:
        data, model = smart_extract(file_bytes, file.content_type)
    except Exception as e:
        raise HTTPException(500, detail=str(e))

    report = PatientReport(patient_name=data.patient_name, birth_date=data.birth_date, report_date=data.report_date)
    db.add(report)
    db.commit()
    db.refresh(report)

    for res in data.results:
        db.add(TestResult(report_id=report.id, test_name=res.test_name, value=res.value, unit=res.unit, min_ref=res.min_ref, max_ref=res.max_ref, confidence_score=res.confidence_score, ai_model_used=model))
    db.commit()

    return {"status": "success", "data": data}
```

```python
import streamlit as st
import requests

def get_health_status(value, min_ref, max_ref):
    if value is None or value == -1: return (None, "#808080", "Info")
    try:
        val = float(value)
        mn = float(min_ref) if min_ref is not None else 0.0
        mx = float(max_ref) if max_ref is not None else float('inf')
    except: return (None, "#808080", "Error")

    if mn == 0.0 and mx != float('inf'): # Case: < 200
        if val <= mx: return (25 + (val/mx)*50, "#2ECC71", "Normal")
        else: return (90, "#E74C3C", "High")
    
    if mn != 0.0 and mx == float('inf'): # Case: > 55
        if val >= mn: return (50, "#2ECC71", "Normal")
        else: return (15, "#E74C3C", "Low")

    try:
        range_span = mx - mn
        if range_span == 0: return (50, "#2ECC71", "Normal")
        relative_pos = (val - mn) / range_span
        score = max(0, min(100, 25 + (relative_pos * 50)))
        
        if 25 <= score <= 75: return (score, "#2ECC71", "Normal")
        elif 15 <= score < 25 or 75 < score <= 85: return (score, "#F1C40F", "Borderline")
        else: return (score, "#E74C3C", "Abnormal")
    except: return (None, "#808080", "Error")

st.set_page_config(page_title="Health AI", layout="wide")
st.title("🧬 Health Trend Tracker")
uploaded_file = st.file_uploader("Upload Report", type=["pdf", "jpg", "png"])

if uploaded_file:
    with st.spinner("AI is analyzing..."):
        files = {"file": (uploaded_file.name, uploaded_file.getvalue(), uploaded_file.type)}
        try:
            res = requests.post("http://localhost:8080/analyze", files=files)
            if res.status_code == 200:
                data = res.json()["data"]
                st.metric("Patient", data.get('patient_name', 'Unknown'))
                st.divider()
                for r in data.get("results", []):
                    score, color, status = get_health_status(r['value'], r.get('min_ref'), r.get('max_ref'))
                    with st.container():
                        c1, c2, c3, c4 = st.columns([2.5, 1, 0.8, 3])
                        c1.markdown(f"**{r['test_name']}**")
                        c1.caption(f"Ref: {r.get('min_ref', 'N/A')} - {r.get('max_ref', 'N/A')}")
                        c2.write(f"**{r['value']}**")
                        c3.caption(r['unit'])
                        if score: 
                            c4.progress(int(score))
                            c4.markdown(f"<span style='color:{color}'>● {status}</span>", unsafe_allow_html=True)
                        st.write("---")
        except Exception as e: st.error(f"Error: {e}")
```
