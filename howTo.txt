3. database.py

Includes the fix to automatically find the certificate on Windows.
Python

import os
import sys
from sqlalchemy import create_engine
from sqlalchemy.ext.declarative import declarative_base
from sqlalchemy.orm import sessionmaker
from dotenv import load_dotenv

load_dotenv()

# 1. Get URL from .env
DATABASE_URL = os.environ.get("DATABASE_URL", "")

# 2. Fix Driver Prefix
# SQLAlchemy needs 'cockroachdb://' to know which dialect to use
if DATABASE_URL.startswith("postgresql://"):
    DATABASE_URL = DATABASE_URL.replace("postgresql://", "cockroachdb://", 1)

# 3. Windows SSL Fix
# If running on Windows, force SQLAlchemy to look in the AppData folder
# for the root.crt file we downloaded in Phase 1.
if sys.platform == "win32" and "sslrootcert" not in DATABASE_URL:
    appdata = os.environ.get("APPDATA")
    if appdata:
        cert_path = os.path.join(appdata, "postgresql", "root.crt")
        # Append the certificate path to the connection string
        DATABASE_URL += f"&sslrootcert={cert_path}"

# 4. Create Engine
# pool_pre_ping checks if the connection is alive before using it
engine = create_engine(DATABASE_URL, pool_pre_ping=True)

SessionLocal = sessionmaker(autocommit=False, autoflush=False, bind=engine)

Base = declarative_base()

def get_db():
    """Dependency for FastAPI"""
    db = SessionLocal()
    try:
        yield db
    finally:
        db.close()

6. extractor.py (Gemini Logic)

Python code

from google import genai
from google.genai import types
import os
from schemas import ExtractionResult

# Initialize the Client
client = genai.Client(api_key=os.environ.get("GEMINI_API_KEY"))

def call_gemini_model(model_name: str, image_bytes: bytes, mime_type: str):
    """
    Helper to call Gemini with strict Pydantic schema enforcement.
    """
    response = client.models.generate_content(
        model=model_name,
        contents=[
            types.Content(
                role="user",
                parts=[
                    # FIX: Instantiate Part directly
                    types.Part(text="Extract blood test data. Rate confidence (0-100)."),
                    
                    # FIX: Instantiate Part directly with inline_data
                    types.Part(inline_data=types.Blob(
                        data=image_bytes, 
                        mime_type=mime_type
                    ))
                ]
            )
        ],
        config={
            "response_mime_type": "application/json",
            "response_schema": ExtractionResult,
        }
    )
    return response.parsed

def smart_extract(image_bytes: bytes, mime_type: str):
    """
    Waterfall Logic: Flash -> Pro
    """
    # Ensure mime_type is valid for Gemini
    valid_mime = mime_type if mime_type in ["application/pdf", "image/png"] else "image/jpeg"
    
    print(f"⚡ Attempting Gemini 2.5 Flash...")
    try:
        # 1. Try Cheap Model (UPDATED TO 2.5)
        data: ExtractionResult = call_gemini_model("gemini-2.5-flash", image_bytes, valid_mime)
        model_used = "gemini-2.5-flash"

        # 2. Calculate Average Confidence
        if data.results:
            avg_conf = sum(r.confidence_score for r in data.results) / len(data.results)
        else:
            avg_conf = 0

        print(f"⚡ Flash Confidence: {avg_conf}%")

        # 3. Fallback Trigger
        if avg_conf < 75 or len(data.results) == 0:
            print(f"⚠️ Low Confidence. Escalating to Gemini 2.5 Pro...")
            # UPDATED TO 2.5 PRO (Stable)
            data = call_gemini_model("gemini-2.5-pro", image_bytes, valid_mime) 
            model_used = "gemini-2.5-pro"

    except Exception as e:
        print(f"❌ Flash Failed: {e}. Retrying with Pro...")
        data = call_gemini_model("gemini-2.5-pro", image_bytes, valid_mime)
        model_used = "gemini-2.5-pro"

    return data, model_used

7. main.py

Python

from fastapi import FastAPI, UploadFile, File, Depends, HTTPException
from sqlalchemy.orm import Session
from database import engine, Base, get_db
from models import PatientReport, TestResult
from extractor import smart_extract
import uvicorn

# Create Tables in CockroachDB if they don't exist
Base.metadata.create_all(bind=engine)

app = FastAPI(title="Health AI")

@app.post("/analyze")
async def analyze(file: UploadFile = File(...), db: Session = Depends(get_db)):
    if file.content_type not in ["application/pdf", "image/jpeg", "image/png"]:
        raise HTTPException(400, detail="Invalid file type")
    
    file_bytes = await file.read()
    
    # AI Extraction
    try:
        data, model = smart_extract(file_bytes, file.content_type)
    except Exception as e:
        raise HTTPException(500, detail=str(e))

    # Save to DB
    report = PatientReport(
        patient_name=data.patient_name,
        birth_date=data.birth_date,
        report_date=data.report_date
    )
    db.add(report)
    db.commit()
    db.refresh(report)

    for res in data.results:
        db.add(TestResult(
            report_id=report.id,
            test_name=res.test_name,
            value=res.value,
            unit=res.unit,
            confidence_score=res.confidence_score,
            ai_model_used=model
        ))
    db.commit()

    return {"status": "success", "data": data, "model": model}

if __name__ == "__main__":
    uvicorn.run(app, host="0.0.0.0", port=8080)

🚀 Phase 3: Run It (Docker)

Open your terminal inside the health_backend folder.

Step 1: Build the Image
Bash

docker build -t health-app .

Step 2: Run the Container Note: We mount the certificate folder so Docker can see it too.
Bash

docker run -d --name health-server -p 8080:8080 --env-file .env -v $env:APPDATA\postgresql:/root/.postgresql health-app

(If the -v command gives an error on Windows, try removing -v $env:APPDATA\postgresql:/root/.postgresql and rely on the database.py logic, but mounting is safer).

Step 3: Test

    Open http://localhost:8080/docs in Chrome. -> for backend

    http://localhost:8501/ -> for dashboard (frontend)

    Click POST /analyze -> Try it out.

    Upload a file.

    Click Execute.

Step 4: If code fails, you have to delete the docker file via terminal, and then rerun from step 1 of building docker image

docker rm -f health-server

If you ran docker restart health-server, it would simply restart the old code. Docker containers are isolated; they don't know you changed a file on your laptop unless you rebuild the image or configure a "Volume mount" (which is complex for beginners).

Rule of Thumb:

    Changed Code (.py files)? → Rebuild (docker build) then Run.

    Changed Env Variables (.env)? → Stop/Start (docker rm then docker run).

    Just crashed/stopped? → Restart (docker restart).
